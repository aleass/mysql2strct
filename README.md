[名字]
host     = ip:port
user     = 账号
pass     = 密码
database = 库
saveDir  = 结构体文件生成目录
pack     = 结构体文件生成的包名,例如model
type     = 连接类型(mysql+ssh 和 tcp)


# (例子)
[yk]
host     = "127.0.0.1:3306"
user     = "root"
pass     = "root"
database = "system"
saveDir  = "D:\\code\\src\\wg-his-data-migrate\\src\\model"
pack     = "model"
type     = "tcp"
